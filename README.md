# candump-py
This script works like the command "candump [port]" from Linux CAN-UTILS

# parameters
You can change thoses differents parameter at the top of the file candump.py :
can_name = 'can0'
bus_type = 'socketcan'
baudrate = '1000000'
human_time = True
add_buffer = False

# how to launch
To launch it, you just have to do :
python3 candump.py
